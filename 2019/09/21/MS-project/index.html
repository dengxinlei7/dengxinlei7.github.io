univariate analysis

data[c(4,5,7,8,11:23,25:27,42:44)] <- lapply(data[c(4,5,7,8,11:23,25:27,42:44)],as.factor)
output=NULL
for(i in 4:ncol(data)){
uni1=coxph(Surv(time,y)~data[,i],data = data)
med=summary(uni1)
b=sprintf(“%.3f”,round(med$conf.int[,1],3))
c=sprintf(“%.3f”,round(med$conf.int[,3],3))
d=sprintf(“%.3f”,round(med$conf.int[,4],3))
pvalue=sprintf(“%.3f”,med$waldtest[3])
e <- paste0(b,”(“,c,”, “,d,”)”)
a <- colnames(data)[i] %>% cbind(.,e) %>%cbind(.,NA)
colnames(a) <- c(‘variables’,’HR(95%CI)’,’pvalue’)
a[,3][nrow(a)] <- pvalue
output=rbind(output,a)
}
write.csv(output,”univariate analysis.csv”,row.names=F)

multivariate analysis

library(mice)
library(norm)
imp <- mice(data[c(2,3,a)],seed = 123,maxit = 50,m=10)
med=Surv(time, y) ~ X1+X2+X3
imp.data <- as.list(1:10)
for (i in 1:10) {
imp.data[[i]] <- complete(imp,action = i)
}
imp.cox <- lapply(imp.data, FUN=function(u){
coxph(formula = med,data = u)})
imp.parmmat <- sapply(imp.cox,coefficients)
minparm <- apply(imp.parmmat,1,min)
maxparm <- apply(imp.parmmat,1,max)
imp.parms <- lapply(imp.cox,coefficients)
imp.ses <- lapply(imp.cox,FUN=function(u){
sqrt(diag(u$var))})
mi.parms <- mi.inference(imp.parms,imp.ses,confidence=0.95)
mi.results <- cbind(exp(mi.parms$est),mi.parms$std.err,mi.parms$lower,mi.parms$upper,
mi.parms$df,mi.parms$signif,minparm,maxparm,mi.parms$fminf)
colnames(mi.results) <- c(“Est”,”StdErr”,”Lower”,”Upper”,”DF”,”P-values”,”Min”,”Max”,”fminf”)
mi.results
write.csv(mi.results,’Multivariate analysis for Y.csv’)
